
namespace = masc

event = {
	id = masc.10000
	hide_window = yes
	is_triggered_only = yes
	fire_only_once = yes

	immediate = {
		set_spawn_system_batch = begin
		# batch-processes the spawn_system effects between "begin" and "end",
		# so caches are recalculated only once rather than for every system spawned
		# can also be used when removing and adding hyperlanes
		no_scope = {
			# makes system positions originate from galactic core
			spawn_system = {
				min_distance >= 0
				max_distance <= 0
				min_orientation_angle = 44
				max_orientation_angle = 46
				initializer = masc_superlargevoid_astless
				hyperlane = no
			}
		}

		random_system = {
			limit = { has_star_flag = masc_the_galaxy_core }
			save_global_event_target_as = masc_the_galaxy_core
		}
		
		# clear all hyperlanes to and within the cluster
		every_system = {
			limit = { has_star_flag = masc_the_galaxy_core }
			isolate_system = yes
		}

		# add hyperlanes
		#event_target:masc_the_galaxy_core = {
		#	add_hyperlane = {
		#		from = event_target:masc_the_galaxy_core
		#		to = event_target:masc_the_galaxy_core
		#	}
		#}
		set_spawn_system_batch = end
		while = {
			count = 20
			random_system = {
			#closest_system = 
				limit = {
					NOT = { has_hyperlane_to = event_target:masc_the_galaxy_core }
				}
				save_global_event_target_as = masc_the_galaxy_core
				
				add_hyperlane = {
					from = this
					to = event_target:masc_the_galaxy_core
				}
			}
		}
		while = {
			count = 0
			event_target:masc_the_galaxy_core = {
				if = {
					limit = {
						any_system = {
							distance = {
								source = prev
								type = euclidean
								min_distance >= 1
								max_distance <= 10
							}
							NOT = { has_hyperlane_to = prev }
						}
					}
					random_system = {
						limit = {
							distance = {
								source = prev
								type = euclidean
								min_distance >= 1
								max_distance <= 10
							}
							NOT = { has_hyperlane_to = prev }
						}
						add_hyperlane = {
							from = this
							to = prev
						}
					}
				}
				else_if = {
					limit = {
						any_system = {
							distance = {
								source = prev
								type = euclidean
								min_distance >= 1
								max_distance <= 15
							}
							NOT = { has_hyperlane_to = prev }
						}
					}
					random_system = {
						limit = {
							distance = {
								source = prev
								type = euclidean
								min_distance >= 1
								max_distance <= 15
							}
							NOT = { has_hyperlane_to = prev }
						}
						add_hyperlane = {
							from = this
							to = prev
						}
					}
				}
				else_if = {
					limit = {
						any_system = {
							distance = {
								source = prev
								type = euclidean
								min_distance >= 1
								max_distance <= 25
							}
							NOT = { has_hyperlane_to = prev }
						}
					}
					random_system = {
						limit = {
							distance = {
								source = prev
								type = euclidean
								min_distance >= 1
								max_distance <= 25
							}
							NOT = { has_hyperlane_to = prev }
						}
						add_hyperlane = {
							from = this
							to = prev
						}
					}
				}
				else_if = {
					limit = {
						any_system = {
							distance = {
								source = prev
								type = euclidean
								min_distance >= 1
								max_distance <= 40
							}
							NOT = { has_hyperlane_to = prev }
						}
					}
					random_system = {
						limit = {
							distance = {
								source = prev
								type = euclidean
								min_distance >= 1
								max_distance <= 40
							}
							NOT = { has_hyperlane_to = prev }
						}
						add_hyperlane = {
							from = this
							to = prev
						}
					}
				}
			}
		}
	}

}
